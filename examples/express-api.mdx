Express REST API server with secrets loaded before the server starts listening. No request can come in before `vault.load()` completes — the server is not bound to a port yet.

## Complete Example

<CodeGroup>

```javascript index.js
import { vault } from '@suronai/sdk'
import express   from 'express'

// Load secrets before anything else — blocks until Telegram approval.
// express() is not called yet, so no requests can come in before secrets are ready.
await vault.load()

// Secrets are now in process.env — safe to use anywhere below.
const app  = express()
const port = process.env.PORT || 3000

app.use(express.json())

// Example: health check — shows the app is running and secrets are loaded
app.get('/health', (req, res) => {
  res.json({
    status:  'ok',
    app:     process.env.VAULT_APP_ID,
    secrets: 'loaded',
  })
})

// Example: a protected route that uses a secret internally (never expose raw values)
app.get('/api/config', (req, res) => {
  // ✅ Use the secret value in logic — don't send it back to the client
  const region      = process.env.REGION      ?? 'us-east-1'
  const featureFlag = process.env.FEATURE_FLAGS === 'true'
  res.json({
    region,           // non-sensitive derived config is fine to expose
    featureFlag,      // boolean derived from secret — not the raw value
    // ❌ never: serviceUrl: process.env.SERVICE_URL  ← leaks secrets to client
  })
})

// Example: using API_KEY from vault for an outbound request
app.post('/api/send', async (req, res) => {
  const { message } = req.body
  if (!message) return res.status(400).json({ error: 'message required' })

  // process.env.API_KEY was decrypted by vault.load() — use it here
  const response = await fetch('https://api.example.com/send', {
    method:  'POST',
    headers: {
      'Content-Type':  'application/json',
      'Authorization': `Bearer ${process.env.API_KEY}`,
    },
    body: JSON.stringify({ message }),
  })

  res.json({ ok: response.ok, status: response.status })
})

app.listen(port, () => {
  console.log(`Server listening on port ${port}`)
})
```

## Key Pattern

```javascript
import { vault } from '@suronai/sdk'
import express   from 'express'

// Load FIRST — server doesn't start until secrets are ready
await vault.load()

const app = express()

app.get('/health', (req, res) => res.json({ status: 'ok' }))

app.listen(3000)
```

<Tip>
  The server only starts listening **after** secrets are loaded. This guarantees no request can access undefined secrets.
</Tip>

## Adding Secrets

Add secrets to your vault using the CLI:

```bash
suron add API_KEY your-api-key-here
suron add SERVICE_URL https://api.yourservice.com
suron add FEATURE_FLAGS enabled
suron add REGION us-east-1
```

## Security Best Practices

<Warning>
  Never expose raw secret values in API responses. Use secrets internally for authentication, configuration, or business logic, but only return derived/non-sensitive data to clients.
</Warning>

**Good:**
```javascript
app.get('/api/config', (req, res) => {
  const featureEnabled = process.env.FEATURE_FLAG === 'true'
  res.json({ feature: featureEnabled })  // boolean, not the raw value
})
```

**Bad:**
```javascript
app.get('/api/config', (req, res) => {
  res.json({ apiKey: process.env.API_KEY })  // ❌ leaks secret to client
})
```
